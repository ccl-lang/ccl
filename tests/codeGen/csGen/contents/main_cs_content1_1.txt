// Main content needed for test function TestCSGenerator1_1 in file csGen_test.go.
// This C# code tests the generated models.

using System;
using CCLGenerated;

class Program
{
    static void Main(string[] args)
    {
        try
        {
            TestUser();
            TestPosition();
            TestEmptyNestedModel();
            TestStrictBinaryParsing();
            Console.WriteLine("Runtime verification successful!");
        }
        catch (Exception e)
        {
            Console.Error.WriteLine($"Verification failed: {e}");
            Environment.Exit(1);
        }
    }

    static void TestUser()
    {
        Console.WriteLine("Testing UserInfo...");
        var user = new UserInfo();
        user.Id = "user123";
        user.Username = "test_user";

        var skin = new SkinInfo();
        skin.Type = 0;

        var basicSkin = new BasicSkin();
        basicSkin.R = 255;
        basicSkin.G = 0;
        basicSkin.B = 0;

        skin.Basic = basicSkin;
        user.Skin = skin;

        // Test serialization
        var userBinary = user.SerializeBinary();
        if (userBinary == null || userBinary.Length == 0)
        {
            throw new Exception("Failed to serialize user");
        }

        // Test deserialization
        var deserializedUser = UserInfo.DeserializeBinary(userBinary);
        if (deserializedUser == null)
        {
            throw new Exception("Failed to deserialize user");
        }

        if (user.Username != "test_user")
        {
            throw new Exception("Username mismatch");
        }
        if (deserializedUser.Username != "test_user")
        {
            throw new Exception("Deserialized username mismatch");
        }

        if (deserializedUser?.Skin?.Basic?.R != 255)
        {
            throw new Exception("Deserialized skin color mismatch");
        }

        var userJson = user.SerializeJson();
        if (!userJson.Contains("\"user_id\""))
        {
            throw new Exception("JsonPropertyName did not apply for user_id");
        }

        var deserializedJsonUser = UserInfo.DeserializeJson(userJson);
        if (deserializedJsonUser == null)
        {
            throw new Exception("Failed to deserialize user JSON");
        }
        if (deserializedJsonUser.Id != "user123")
        {
            throw new Exception("JSON user id mismatch");
        }
        if (deserializedJsonUser?.Skin?.Basic?.R != 255)
        {
            throw new Exception("JSON nested skin color mismatch");
        }

        Console.WriteLine("UserInfo test passed!");
    }

    static void TestPosition()
    {
        Console.WriteLine("Testing PositionInfo...");
        var pos = new PositionInfo();
        pos.X = 10;
        pos.Y = 20;
        pos.Z = 30;

        if (pos.X != 10)
        {
            throw new Exception("PositionInfo X mismatch");
        }

        // Test serialization
        var posBinary = pos.SerializeBinary();
        if (posBinary == null || posBinary.Length == 0)
        {
            throw new Exception("Failed to serialize position");
        }

        // Test deserialization
        var deserializedPos = PositionInfo.DeserializeBinary(posBinary);
        if (deserializedPos == null)
        {
            throw new Exception("Failed to deserialize position");
        }

        if (deserializedPos.X != 10)
        {
            throw new Exception("Deserialized PositionInfo X mismatch");
        }

        Console.WriteLine("PositionInfo test passed!");
    }

    static void TestEmptyNestedModel()
    {
        Console.WriteLine("Testing empty nested model binary...");
        var holder = new EmptyModelHolder();
        holder.Empty = new EmptyModel();
        holder.EmptyItems.Add(new EmptyModel());

        var deserializedHolder = EmptyModelHolder.DeserializeBinary(holder.SerializeBinary());
        if (deserializedHolder == null)
        {
            throw new Exception("Empty model holder returned null");
        }
        if (deserializedHolder.Empty == null)
        {
            throw new Exception("Empty nested model became null");
        }
        if (deserializedHolder.EmptyItems.Count != 1)
        {
            throw new Exception("Empty model array length mismatch");
        }
        if (deserializedHolder.EmptyItems[0] == null)
        {
            throw new Exception("Empty model array item became null");
        }

        Console.WriteLine("Empty nested model test passed!");
    }

    static void TestStrictBinaryParsing()
    {
        Console.WriteLine("Testing StrictBinaryParsing...");
        var strictUserData = new StrictUserData();
        strictUserData.Level = 7;
        strictUserData.Name = "Old Save";
        var oldData = strictUserData.SerializeBinary();

        var looseUserData = LooseUserData.DeserializeBinary(oldData);
        if (looseUserData == null)
        {
            throw new Exception("Loose binary parser returned null");
        }
        if (looseUserData.Level != 7)
        {
            throw new Exception("Loose binary parser level mismatch");
        }
        if (looseUserData.Name != "Old Save")
        {
            throw new Exception("Loose binary parser name mismatch");
        }
        if (looseUserData.LastQuestDone != 0)
        {
            throw new Exception("Loose binary parser default field mismatch");
        }

        Console.WriteLine("StrictBinaryParsing test passed!");
    }
}
