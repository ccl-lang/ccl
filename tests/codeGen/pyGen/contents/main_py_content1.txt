# Main content needed for test function TestPythonGenerator1 in file pyGen1_test.go.

import sys
import os
import json

# Add the current directory to sys.path to ensure we can import the generated models
sys.path.append(os.getcwd())

from models import (
	UserInfo,
	SkinInfo,
	BasicSkin,
	PositionInfo,
	UsersList,
	LooseUserData,
	StrictUserData,
	EmptyModel,
	EmptyModelHolder,
)

def test_user():
	print("Testing UserInfo...")
	user = UserInfo()
	user.id = "user123"
	user.username = "test_user"
	
	skin = SkinInfo()
	skin.type = 0
	
	basic_skin = BasicSkin()
	basic_skin.r = 255
	basic_skin.g = 0
	basic_skin.b = 0
	
	skin.basic = basic_skin
	user.skin = skin

	all_users = UsersList()
	all_users.users = []
	all_users.users.append(user)
	other_users = UsersList.deserialize_binary(all_users.serialize_binary())
	if len(other_users.users) != len(all_users.users):
		raise ValueError("length of users are not equal!")

	# Test serialization
	user_binary = user.serialize_binary()
	if not user_binary:
		raise Exception("Failed to serialize user")

	# Test deserialization
	deserialized_user = UserInfo.deserialize_binary(user_binary)
	if not deserialized_user:
		raise Exception("Failed to deserialize user")

	if user.username != "test_user":
		raise Exception("Username mismatch")
	if deserialized_user.username != "test_user":
		raise Exception("Deserialized username mismatch")
	
	if deserialized_user.skin.basic.r != 255:
		raise Exception("Deserialized skin color mismatch")

	user_json = user.serialize_json()
	user_json_data = json.loads(user_json)
	if "user_id" not in user_json_data:
		raise Exception("JsonPropertyName did not apply for user_id")

	deserialized_json_user = UserInfo.deserialize_json(user_json)
	if not deserialized_json_user:
		raise Exception("Failed to deserialize user JSON")

	if deserialized_json_user.id != "user123":
		raise Exception("JSON user id mismatch")
	if deserialized_json_user.skin.basic.r != 255:
		raise Exception("JSON nested skin color mismatch")

	print("UserInfo test passed!")

def test_position():
	print("Testing PositionInfo...")
	pos = PositionInfo()
	pos.x = 10
	pos.y = 20
	pos.z = 30

	if pos.x != 10:
		raise Exception("PositionInfo X mismatch")

	# Test serialization
	pos_binary = pos.serialize_binary()
	if not pos_binary:
		raise Exception("Failed to serialize position")

	# Test deserialization
	deserialized_pos = PositionInfo.deserialize_binary(pos_binary)
	if not deserialized_pos:
		raise Exception("Failed to deserialize position")

	if deserialized_pos.x != 10:
		raise Exception("Deserialized PositionInfo X mismatch")

	print("PositionInfo test passed!")

def test_empty_nested_model():
	print("Testing empty nested model binary...")
	holder = EmptyModelHolder()
	holder.empty = EmptyModel()
	holder.empty_items = [EmptyModel()]

	deserialized_holder = EmptyModelHolder.deserialize_binary(holder.serialize_binary())
	if deserialized_holder is None:
		raise Exception("Empty model holder returned None")
	if deserialized_holder.empty is None:
		raise Exception("Empty nested model became None")
	if len(deserialized_holder.empty_items) != 1:
		raise Exception("Empty model array length mismatch")
	if deserialized_holder.empty_items[0] is None:
		raise Exception("Empty model array item became None")

	print("Empty nested model test passed!")

def test_deep_clone():
	print("Testing DeepClone...")
	user = UserInfo()
	user.id = "clone-user"
	user.username = "clone_source"

	skin = SkinInfo()
	basic_skin = BasicSkin()
	basic_skin.r = 255
	basic_skin.g = 1
	basic_skin.b = 2
	skin.basic = basic_skin
	user.skin = skin

	users_list = UsersList()
	users_list.users = [user]
	users_list.usernames = ["alpha", "beta"]
	users_list.photos = [bytes([1, 2, 3])]

	clone = users_list.deep_clone()
	if clone is users_list:
		raise Exception("DeepClone returned the original model")
	if clone.users is users_list.users or clone.users[0] is users_list.users[0]:
		raise Exception("DeepClone reused nested users")
	if clone.users[0].skin is users_list.users[0].skin:
		raise Exception("DeepClone reused nested skin")

	clone.users[0].skin.basic.r = 99
	if users_list.users[0].skin.basic.r != 255:
		raise Exception("DeepClone nested mutation affected original")
	clone.usernames[0] = "changed"
	if users_list.usernames[0] != "alpha":
		raise Exception("DeepClone primitive array mutation affected original")
	if clone.photos is users_list.photos:
		raise Exception("DeepClone reused bytes array")

	print("DeepClone test passed!")

def test_strict_binary_parsing():
	print("Testing StrictBinaryParsing...")
	strict_user_data = StrictUserData()
	strict_user_data.level = 7
	strict_user_data.name = "Old Save"
	old_data = strict_user_data.serialize_binary()

	loose_user_data = LooseUserData.deserialize_binary(old_data)
	if loose_user_data is None:
		raise Exception("Loose binary parser returned None")
	if loose_user_data.level != 7:
		raise Exception("Loose binary parser level mismatch")
	if loose_user_data.name != "Old Save":
		raise Exception("Loose binary parser name mismatch")
	if loose_user_data.last_quest_done != 0:
		raise Exception("Loose binary parser default field mismatch")

	print("StrictBinaryParsing test passed!")

if __name__ == "__main__":
	try:
		test_user()
		test_position()
		test_empty_nested_model()
		test_deep_clone()
		test_strict_binary_parsing()
		print("Runtime verification successful!")
	except Exception as e:
		print(f"Verification failed: {e}")
		sys.exit(1)
