import { UserInfo } from './models/UserInfo';
import { SkinInfo } from './models/SkinInfo';
import { BasicSkin } from './models/BasicSkin';
import { PositionInfo } from './models/PositionInfo';
import { LooseUserData } from './models/LooseUserData';

function testUser() {
	console.log("Testing UserInfo...");
	const user = new UserInfo();
	user.id = "user123";
	user.username = "test_user";

	const skin = new SkinInfo();
	skin.type = 0;

	const basicSkin = new BasicSkin();
	basicSkin.r = 255;
	basicSkin.g = 0;
	basicSkin.b = 0;

	skin.basic = basicSkin;
	user.skin = skin;

	// Test serialization
	const userBinary = user.serializeBinary();
	if (!userBinary || userBinary.length === 0) {
		throw new Error("Failed to serialize user");
	}

	// Test deserialization
	const deserializedUser = UserInfo.deserializeBinary(userBinary);
	if (!deserializedUser) {
		throw new Error("Failed to deserialize user");
	}

	if (user.username !== "test_user") {
		throw new Error("Username mismatch");
	}
	if (deserializedUser.username !== "test_user") {
		throw new Error("Deserialized username mismatch");
	}

	if (deserializedUser?.skin?.basic?.r !== 255) {
		throw new Error("Deserialized skin color mismatch");
	}

	const userJson = user.serializeJson();
	const userJsonData = JSON.parse(userJson);
	if (!Object.prototype.hasOwnProperty.call(userJsonData, "user_id")) {
		throw new Error("JsonPropertyName did not apply for user_id");
	}

	const deserializedJsonUser = UserInfo.deserializeJson(userJson);
	if (!deserializedJsonUser) {
		throw new Error("Failed to deserialize user JSON");
	}
	if (deserializedJsonUser.id !== "user123") {
		throw new Error("JSON user id mismatch");
	}
	if (deserializedJsonUser?.skin?.basic?.r !== 255) {
		throw new Error("JSON nested skin color mismatch");
	}

	console.log("UserInfo test passed!");
}

function testEmptyNestedModel() {
	console.log("Testing empty nested model binary...");
	const holder = new EmptyModelHolder();
	holder.empty = new EmptyModel();
	holder.emptyItems = [new EmptyModel()];

	const deserializedHolder = EmptyModelHolder.deserializeBinary(holder.serializeBinary());
	if (!deserializedHolder) {
		throw new Error("Empty model holder returned null");
	}
	if (!deserializedHolder.empty) {
		throw new Error("Empty nested model became null");
	}
	if (deserializedHolder.emptyItems.length !== 1) {
		throw new Error("Empty model array length mismatch");
	}
	if (!deserializedHolder.emptyItems[0]) {
		throw new Error("Empty model array item became null");
	}

	console.log("Empty nested model test passed!");
}

function testDeepClone() {
	console.log("Testing DeepClone...");
	const user = new UserInfo();
	user.id = "clone-user";
	user.username = "clone_source";

	const skin = new SkinInfo();
	const basicSkin = new BasicSkin();
	basicSkin.r = 255;
	basicSkin.g = 1;
	basicSkin.b = 2;
	skin.basic = basicSkin;
	user.skin = skin;

	const usersList = new UsersList();
	usersList.users = [user];
	usersList.usernames = ["alpha", "beta"];
	usersList.photos = [new Uint8Array([1, 2, 3])];

	const clone = usersList.deepClone();
	if (clone === usersList) {
		throw new Error("DeepClone returned the original model");
	}
	if (clone.users === usersList.users || clone.users[0] === usersList.users[0]) {
		throw new Error("DeepClone reused nested users");
	}
	if (clone.users[0]?.skin === usersList.users[0]?.skin) {
		throw new Error("DeepClone reused nested skin");
	}

	clone.users[0]!.skin!.basic!.r = 99;
	if (usersList.users[0]!.skin!.basic!.r !== 255) {
		throw new Error("DeepClone nested mutation affected original");
	}
	clone.usernames[0] = "changed";
	if (usersList.usernames[0] !== "alpha") {
		throw new Error("DeepClone primitive array mutation affected original");
	}
	clone.photos[0][0] = 9;
	if (usersList.photos[0][0] !== 1) {
		throw new Error("DeepClone bytes array mutation affected original");
	}

	console.log("DeepClone test passed!");
}

function testStrictBinaryParsing() {
	console.log("Testing StrictBinaryParsing...");
	const strictUserData = new StrictUserData();
	strictUserData.level = 7;
	strictUserData.name = "Old Save";
	const oldData = strictUserData.serializeBinary();

	const looseUserData = LooseUserData.deserializeBinary(oldData);
	if (!looseUserData) {
		throw new Error("Loose binary parser returned null");
	}
	if (looseUserData.level !== 7) {
		throw new Error("Loose binary parser level mismatch");
	}
	if (looseUserData.name !== "Old Save") {
		throw new Error("Loose binary parser name mismatch");
	}
	if (looseUserData.lastQuestDone !== 0) {
		throw new Error("Loose binary parser default field mismatch");
	}

	console.log("StrictBinaryParsing test passed!");
}
