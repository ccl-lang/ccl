// Main content needed for test function TestGoGenerator1 in file goGen1_test.go.

package main

import (
	"ccl_test_generated/models"
	"fmt"
	"strings"
)

func main() {
	check := func(condition bool, message string) {
		if !condition {
			panic(message)
		}
	}

	// Test creating a UserInfo
	user := models.UserInfo{
		Id:       "user123",
		Username: "test_user",
		Skin: &models.SkinInfo{
			Type: 0,
			Basic: &models.BasicSkin{
				R: 255,
				G: 0,
				B: 0,
			},
			Texture: &models.TextureSkin{
				Texture:      []string{},
				SomeInts:     []int{},
				SomeIntegers: []int64{},
			},
		},
	}

	userBinary, err := user.SerializeBinary()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize user: %v", err))
	}

	deserializedUser := &models.UserInfo{}
	err = deserializedUser.DeserializeBinary(userBinary)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize user: %v", err))
	}

	check(user.Username == "test_user", "Username mismatch")
	check(deserializedUser.Username == "test_user", "Deserialized username mismatch")

	userJSON, err := user.SerializeJSON()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize user JSON: %v", err))
	}
	check(strings.Contains(userJSON, "\"user_id\""), "JsonPropertyName did not apply for user_id")

	deserializedJSONUser := &models.UserInfo{}
	err = deserializedJSONUser.DeserializeJSON(userJSON)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize user JSON: %v", err))
	}
	check(deserializedJSONUser.Id == user.Id, "JSON user id mismatch")
	check(deserializedJSONUser.Skin != nil, "JSON user skin is nil")
	check(deserializedJSONUser.Skin.Basic != nil, "JSON user basic skin is nil")
	check(deserializedJSONUser.Skin.Basic.R == 255, "JSON nested skin mismatch")

	// Test creating a PositionInfo
	pos := models.PositionInfo{
		X: 10,
		Y: 20,
		Z: 30,
	}

	check(pos.X == 10, "PositionInfo X mismatch")

	positionBinary, err := pos.SerializeBinary()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize position: %v", err))
	}

	deserializedPosition := &models.PositionInfo{}
	err = deserializedPosition.DeserializeBinary(positionBinary)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize position: %v", err))
	}

	check(pos.X == 10, "PositionInfo X mismatch")

	// Test TextureSkin with arrays
	texture := models.TextureSkin{
		Texture:      []string{"a.png", "b.png", "c.png"},
		SomeInts:     []int{1, 2, 3, 4, 5},
		SomeIntegers: []int64{1, 2, 3, 4, 5},
	}
	textureBinary, err := texture.SerializeBinary()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize texture skin: %v", err))
	}

	deserializedTexture := &models.TextureSkin{}
	err = deserializedTexture.DeserializeBinary(textureBinary)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize texture skin: %v", err))
	}

	check(len(deserializedTexture.Texture) == len(texture.Texture), "Texture array length mismatch")
	for i, item := range texture.Texture {
		check(deserializedTexture.Texture[i] == item, "Texture array item mismatch")
	}
	check(len(deserializedTexture.SomeInts) == len(texture.SomeInts), "SomeInts array length mismatch")
	for i, item := range texture.SomeInts {
		check(deserializedTexture.SomeInts[i] == item, "SomeInts array item mismatch")
	}
	check(len(deserializedTexture.SomeIntegers) == len(texture.SomeIntegers), "SomeIntegers array length mismatch")
	for i, item := range texture.SomeIntegers {
		check(deserializedTexture.SomeIntegers[i] == item, "SomeIntegers array item mismatch")
	}

	// Test UsersList with arrays of custom models
	user2 := &models.UserInfo{
		Id:       "user456",
		Username: "second_user",
		Skin: &models.SkinInfo{
			Type: 1,
			Basic: &models.BasicSkin{
				R: 10,
				G: 20,
				B: 30,
			},
			Texture: &models.TextureSkin{
				Texture:      []string{"t1.png"},
				SomeInts:     []int{9, 8},
				SomeIntegers: []int64{9, 8},
			},
		},
	}
	user3 := &models.UserInfo{
		Id:       "user789",
		Username: "third_user",
		Skin: &models.SkinInfo{
			Type: 2,
			Basic: &models.BasicSkin{
				R: 1,
				G: 2,
				B: 3,
			},
			Texture: &models.TextureSkin{
				Texture:      []string{"t2.png", "t3.png"},
				SomeInts:     []int{7},
				SomeIntegers: []int64{7},
			},
		},
	}

	usersList := models.UsersList{
		Users:     []*models.UserInfo{&user, user2},
		MoreUsers: []*models.UserInfo{user3},
	}

	usersListBinary, err := usersList.SerializeBinary()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize users list: %v", err))
	}

	deserializedUsersList := &models.UsersList{}
	err = deserializedUsersList.DeserializeBinary(usersListBinary)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize users list: %v", err))
	}

	check(len(deserializedUsersList.Users) == len(usersList.Users), "UsersList.Users length mismatch")
	check(len(deserializedUsersList.MoreUsers) == len(usersList.MoreUsers), "UsersList.MoreUsers length mismatch")

	usersListJSON, err := usersList.SerializeJSON()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize users list JSON: %v", err))
	}
	deserializedJSONUsersList := &models.UsersList{}
	err = deserializedJSONUsersList.DeserializeJSON(usersListJSON)
	if err != nil {
		panic(fmt.Sprintf("Failed to deserialize users list JSON: %v", err))
	}
	check(len(deserializedJSONUsersList.Users) == len(usersList.Users), "JSON UsersList.Users length mismatch")
	check(deserializedJSONUsersList.Users[1].Skin.Texture.Texture[0] == "t1.png", "JSON UsersList nested array mismatch")

	check(deserializedUsersList.Users[0].Id == usersList.Users[0].Id, "UsersList.Users[0].Id mismatch")
	check(deserializedUsersList.Users[0].Username == usersList.Users[0].Username, "UsersList.Users[0].Username mismatch")
	check(deserializedUsersList.Users[0].Skin != nil, "UsersList.Users[0].Skin is nil")
	check(deserializedUsersList.Users[0].Skin.Basic != nil, "UsersList.Users[0].Skin.Basic is nil")
	check(deserializedUsersList.Users[0].Skin.Texture != nil, "UsersList.Users[0].Skin.Texture is nil")

	check(deserializedUsersList.Users[1].Id == usersList.Users[1].Id, "UsersList.Users[1].Id mismatch")
	check(deserializedUsersList.Users[1].Username == usersList.Users[1].Username, "UsersList.Users[1].Username mismatch")
	check(deserializedUsersList.Users[1].Skin != nil, "UsersList.Users[1].Skin is nil")
	check(deserializedUsersList.Users[1].Skin.Basic != nil, "UsersList.Users[1].Skin.Basic is nil")
	check(deserializedUsersList.Users[1].Skin.Texture != nil, "UsersList.Users[1].Skin.Texture is nil")
	check(len(deserializedUsersList.Users[1].Skin.Texture.Texture) == len(usersList.Users[1].Skin.Texture.Texture), "UsersList.Users[1].Skin.Texture.Texture length mismatch")
	check(deserializedUsersList.Users[1].Skin.Texture.Texture[0] == usersList.Users[1].Skin.Texture.Texture[0], "UsersList.Users[1].Skin.Texture.Texture[0] mismatch")

	check(deserializedUsersList.MoreUsers[0].Id == usersList.MoreUsers[0].Id, "UsersList.MoreUsers[0].Id mismatch")
	check(deserializedUsersList.MoreUsers[0].Username == usersList.MoreUsers[0].Username, "UsersList.MoreUsers[0].Username mismatch")
	check(deserializedUsersList.MoreUsers[0].Skin != nil, "UsersList.MoreUsers[0].Skin is nil")
	check(deserializedUsersList.MoreUsers[0].Skin.Basic != nil, "UsersList.MoreUsers[0].Skin.Basic is nil")
	check(deserializedUsersList.MoreUsers[0].Skin.Texture != nil, "UsersList.MoreUsers[0].Skin.Texture is nil")

	strictUserData := models.StrictUserData{
		Level: 7,
		Name:  "Old Save",
	}
	oldSaveData, err := strictUserData.SerializeBinary()
	if err != nil {
		panic(fmt.Sprintf("Failed to serialize strict user data: %v", err))
	}

	looseUserData := &models.LooseUserData{}
	err = looseUserData.DeserializeBinary(oldSaveData)
	if err != nil {
		panic(fmt.Sprintf("Loose binary parser should not return error: %v", err))
	}
	check(looseUserData.Level == 7, "Loose binary parser level mismatch")
	check(looseUserData.Name == "Old Save", "Loose binary parser name mismatch")
	check(looseUserData.LastQuestDone == 0, "Loose binary parser default field mismatch")

	fmt.Println("Runtime verification successful!")
}
