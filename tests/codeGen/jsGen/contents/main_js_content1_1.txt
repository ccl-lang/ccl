import { UserInfo } from './models/UserInfo.js';
import { SkinInfo } from './models/SkinInfo.js';
import { BasicSkin } from './models/BasicSkin.js';
import { PositionInfo } from './models/PositionInfo.js';
import { UsersList } from './models/UsersList.js';

function testUser() {
	console.log("Testing UserInfo...");
	const user = new UserInfo();
	user.id = "user123";
	user.username = "test_user";

	const skin = new SkinInfo();
	skin.type = 0;

	const basicSkin = new BasicSkin();
	basicSkin.r = 255;
	basicSkin.g = 0;
	basicSkin.b = 0;

	skin.basic = basicSkin;
	user.skin = skin;

	if (user.username !== "test_user") {
		throw new Error("Username mismatch");
	}
	if (user.skin.basic.r !== 255) {
		throw new Error("Skin color mismatch");
	}

	const userJson = user.serializeJson();
	const userJsonData = JSON.parse(userJson);
	if (!Object.prototype.hasOwnProperty.call(userJsonData, "user_id")) {
		throw new Error("JsonPropertyName did not apply for user_id");
	}

	const deserializedJsonUser = UserInfo.deserializeJson(userJson);
	if (!deserializedJsonUser) {
		throw new Error("Failed to deserialize user JSON");
	}
	if (deserializedJsonUser.id !== "user123") {
		throw new Error("JSON user id mismatch");
	}
	if (deserializedJsonUser?.skin?.basic?.r !== 255) {
		throw new Error("JSON nested skin color mismatch");
	}

	console.log("UserInfo test passed!");
}

function testPosition() {
	console.log("Testing PositionInfo...");
	const pos = new PositionInfo();
	pos.x = 10;
	pos.y = 20;
	pos.z = 30;

	if (pos.x !== 10) {
		throw new Error("PositionInfo X mismatch");
	}

	console.log("PositionInfo test passed!");
}

function testDeepClone() {
	console.log("Testing DeepClone...");
	const user = new UserInfo();
	user.id = "clone-user";
	user.username = "clone_source";

	const skin = new SkinInfo();
	const basicSkin = new BasicSkin();
	basicSkin.r = 255;
	basicSkin.g = 1;
	basicSkin.b = 2;
	skin.basic = basicSkin;
	user.skin = skin;

	const usersList = new UsersList();
	usersList.users = [user];
	usersList.usernames = ["alpha", "beta"];
	usersList.photos = [new Uint8Array([1, 2, 3])];

	const clone = usersList.deepClone();
	if (clone === usersList) {
		throw new Error("DeepClone returned the original model");
	}
	if (clone.users === usersList.users || clone.users[0] === usersList.users[0]) {
		throw new Error("DeepClone reused nested users");
	}
	if (clone.users[0].skin === usersList.users[0].skin) {
		throw new Error("DeepClone reused nested skin");
	}

	clone.users[0].skin.basic.r = 99;
	if (usersList.users[0].skin.basic.r !== 255) {
		throw new Error("DeepClone nested mutation affected original");
	}
	clone.usernames[0] = "changed";
	if (usersList.usernames[0] !== "alpha") {
		throw new Error("DeepClone primitive array mutation affected original");
	}
	clone.photos[0][0] = 9;
	if (usersList.photos[0][0] !== 1) {
		throw new Error("DeepClone bytes array mutation affected original");
	}

	console.log("DeepClone test passed!");
}

try {
	testUser();
	testPosition();
	testDeepClone();
	console.log("Runtime verification successful!");
} catch (e) {
	console.error("Verification failed:", e);
	process.exit(1);
}
