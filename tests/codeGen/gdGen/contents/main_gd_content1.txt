# Main content needed for test function TestGdGenerator1 in file gdGen1_test.go.
extends SceneTree

func _init():
	print("Starting GDScript runtime verification...")
	
	var user_script = load("res://models/user_info.gd")
	if user_script == null:
		print("Error: Could not load user_info.gd")
		quit(1)
		return

	var user = user_script.new()
	user.id = "user123"
	user.username = "test_user"
	
	# Test SkinInfo
	var skin_info_script = load("res://models/skin_info.gd")
	var skin_info = skin_info_script.new()
	skin_info.type = SkinType.SkinTypeEnum.BASIC

	var basic_skin_script = load("res://models/basic_skin.gd")
	var basic_skin = basic_skin_script.new()
	basic_skin.r = 255
	basic_skin.g = 0
	basic_skin.b = 0

	skin_info.basic = basic_skin
	user.skin = skin_info

	if user.username != "test_user":
		print("Error: Username mismatch")
		quit(1)
		return

	# Test binary serialization/deserialization
	var user_bin = user.serialize_binary()
	var user_from_bin = user_script.deserialize_binary(user_bin)
	if user_from_bin == null:
		print("Error: deserialize_binary returned null")
		quit(1)
		return
	if user_from_bin.username != "test_user":
		print("Error: Binary username mismatch")
		quit(1)
		return
	if user_from_bin.skin == null or user_from_bin.skin.basic == null or user_from_bin.skin.basic.r != 255:
		print("Error: Binary nested model mismatch")
		quit(1)
		return

	# Test JSON serialization/deserialization
	var user_json = user.serialize_json()
	var user_from_json = user_script.deserialize_json(user_json)
	if user_from_json == null:
		print("Error: deserialize_json returned null")
		quit(1)
		return
	if user_from_json.username != "test_user":
		print("Error: JSON username mismatch")
		quit(1)
		return
	if user_from_json.skin == null or user_from_json.skin.basic == null or user_from_json.skin.basic.r != 255:
		print("Error: JSON nested model mismatch")
		quit(1)
		return

	var json_checker = JSON.new()
	if json_checker.parse(user_json) != OK:
		print("Error: JSON parse failed for user_json")
		quit(1)
		return
	if typeof(json_checker.data) != TYPE_DICTIONARY:
		print("Error: JSON result is not a dictionary")
		quit(1)
		return
	if not json_checker.data.has("user_id"):
		print("Error: JsonPropertyName did not apply for user_id")
		quit(1)
		return
	if json_checker.data["user_id"] != "user123":
		print("Error: JsonPropertyName value mismatch")
		quit(1)
		return

	# Test PositionInfo (direct access)
	var pos = PositionInfo.new()
	pos.x = 10
	pos.y = 20
	pos.z = 30

	if pos.x != 10:
		print("Error: PositionInfo X mismatch")
		quit(1)
		return

	var game_item = GameItem.new()
	game_item.item_type = GameItem.GameItemType.GOLD
	game_item.count = 50

	var game_shop_item = GameShopItem.new()
	game_shop_item.featured_type = GameItem.GameItemType.GEM
	if game_item.item_type != GameItem.GameItemType.GOLD or game_shop_item.featured_type != GameItem.GameItemType.GEM:
		print("Error: nested enum reference mismatch")
		quit(1)
		return

	var empty_holder = EmptyModelHolder.new()
	empty_holder.empty = EmptyModel.new()
	empty_holder.empty_items.append(EmptyModel.new())
	var empty_holder_from_bin = EmptyModelHolder.deserialize_binary(empty_holder.serialize_binary())
	if empty_holder_from_bin == null:
		print("Error: empty model holder returned null")
		quit(1)
		return
	if empty_holder_from_bin.empty == null:
		print("Error: empty nested model became null")
		quit(1)
		return
	if empty_holder_from_bin.empty_items.size() != 1:
		print("Error: empty model array length mismatch")
		quit(1)
		return
	if empty_holder_from_bin.empty_items[0] == null:
		print("Error: empty model array item became null")
		quit(1)
		return

	var clone_users = UsersList.new()
	clone_users.users.append(user)
	clone_users.usernames.append("alpha")
	var clone_photo := PackedByteArray([1, 2, 3])
	clone_users.photos.append(clone_photo)
	var clone_result = clone_users.deep_clone()
	if clone_result == clone_users:
		print("Error: DeepClone returned the original model")
		quit(1)
		return
	if clone_result.users == clone_users.users or clone_result.users[0] == clone_users.users[0]:
		print("Error: DeepClone reused nested users")
		quit(1)
		return
	if clone_result.users[0].skin == clone_users.users[0].skin:
		print("Error: DeepClone reused nested skin")
		quit(1)
		return
	clone_result.users[0].skin.basic.r = 99
	if clone_users.users[0].skin.basic.r != 255:
		print("Error: DeepClone nested mutation affected original")
		quit(1)
		return
	clone_result.usernames[0] = "changed"
	if clone_users.usernames[0] != "alpha":
		print("Error: DeepClone primitive array mutation affected original")
		quit(1)
		return
	clone_result.photos[0][0] = 9
	if clone_users.photos[0][0] != 1:
		print("Error: DeepClone bytes array mutation affected original")
		quit(1)
		return

	var strict_user_data = StrictUserData.new()
	strict_user_data.level = 7
	strict_user_data.name = "Old Save"
	var old_data = strict_user_data.serialize_binary()

	var loose_user_data = LooseUserData.deserialize_binary(old_data)
	if loose_user_data == null:
		print("Error: loose binary parser returned null")
		quit(1)
		return
	if loose_user_data.level != 7 or loose_user_data.name != "Old Save" or loose_user_data.last_quest_done != 0:
		print("Error: loose binary parser partial model mismatch")
		quit(1)
		return

	print("Runtime verification successful!")
	quit(0)

func _ready():
	quit(1) # Should not reach here
